import numpy as np
import scipy.special as sp
import pkuplot

x = np.linspace(-1.0, 1.0, 100)

3. Associated Legendre polynomials of the 1st kind¶

In [3]:
def AssoLegendre(m,n,x):
    '''
    n-degree(base)
    m-order(shoulder)
    计算 Associated Legendre 函数
    并将其结果标准化（值域缩减为1以内） by * sqrt( factorial(n-m)/factorial(n+m) )
    '''
    # Associated Legendre
    y = sp.lpmv(m,n,x)
    # Normalize it
    y = y * np.sqrt(np.math.factorial(n-m)/np.math.factorial(n+m))
    return y
