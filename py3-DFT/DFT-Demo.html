def tr_syn(C,keep_to_wavenumber):
    '''
    Truncate off, then Synthesis back
    '''
    Ct = C.copy()
    Ct[keep_to_wavenumber+1:] = 0
    return np.fft.irfft(Ct)

z500_0 = tr_syn(C,0)
z500_1 = tr_syn(C,1)
z500_2 = tr_syn(C,2)
z500_3 = tr_syn(C,3)
z500_7 = tr_syn(C,7)

# plot
pkuplot.plot2d(6,[lon,lon,lon,lon,lon,lon],[z500,z500_0,z500_1,z500_2,z500_3,z500_7],
               style=['c-','y--','k-','r-','b-','g--'],
               lw=[10,2,2,2,2,2],
               lb=['raw','just W#0','to W#1','to W#2','to W#3','to W#7'],
               ti='2016-Jan-21',xl='Lon',yl='Z500 (gpm)',xlim=[-30,390])

def wavenumber(C,wn):
    '''
    Only keep one single wavenumber, then Synthesis back
    '''
    Ct = C.copy()*0
    Ct[wn] = C[wn]

    return np.fft.irfft(Ct)

z500_0 = wavenumber(C,0)
z500_1 = wavenumber(C,1)
z500_2 = wavenumber(C,2)
z500_3 = wavenumber(C,3)
z500_7 = wavenumber(C,7)

# plot
pkuplot.plot2d(4,[lon,lon,lon,lon],[z500_1,z500_2,z500_3,z500_7],
               style=['k-','r-','b-','g--'],
               lw=[2,2,2,2],
               lb=['just W#1','just W#2','just W#3','just W#7'],
